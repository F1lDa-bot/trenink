            function setupFilters() {

                const buttons = ['Vše', ...categories];

                filterButtonsContainer.innerHTML = buttons.map(category => 

                    `<button class="filter-btn px-3 py-1.5 text-sm rounded-md transition-colors duration-200 ${category === 'Vše' ? 'active' : 'bg-slate-200 hover:bg-slate-300 text-slate-700'}" data-category="${category}">${category}</button>`

                ).join('');


                filterButtonsContainer.addEventListener('click', (e) => {

                    if (e.target.tagName === 'BUTTON') {

                        activeFilter = e.target.dataset.category;

                        document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active', 'bg-sky-600', 'text-white'));

                        document.querySelectorAll('.filter-btn').forEach(btn => {

                            if(btn.dataset.category !== activeFilter) {

                                btn.classList.add('bg-slate-200', 'hover:bg-slate-300', 'text-slate-700')

                            }

                        });

                        e.target.classList.add('active');

                        e.target.classList.remove('bg-slate-200', 'hover:bg-slate-300', 'text-slate-700');

                        renderCards();

                    }

                });

            }
